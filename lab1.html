def modularity(G, L):
    total = 0
    nodes = list(G.nodes())
    A = nx.adjacency_matrix(G).todense()
    for i in range(A.shape[0]):
        for j in range(A.shape[1]):
            pij = G.degree(nodes[i]) * G.degree(nodes[j]) / (2 * L)
            total += A.item((i, j)) - pij
    return total / (2 * L)

def detect_communities(G):
    max_components = []
    max_M = -math.inf
    nodes = G.nodes()
    components = []
    L = G.number_of_edges()
    while (len(components) != len(nodes)):
        ebcs = nx.edge_betweenness_centrality(G)
        max_e = max(ebcs, key=ebcs.get)
        G.remove_edge(max_e[0], max_e[1])

        M = 0        
        for SG in components:
            M += modularity(SG, L)
            
        components = list(nx.connected_component_subgraphs(G))
#         print(M, len(components))

        if (M > max_M):
            max_M = M
            max_components = components
    return max_M, max_components
